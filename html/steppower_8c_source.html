306{
307 // If L2 or R2 not pressed, powerup is unchanged.
308 if ((pjoy->grfbtnPressed & (PAD_L2 | PAD_R2)) == _NOT_PRESSED)
309 {
310 return iCur;
311 }
312
313 // Default to next powerup (assume R2 pressed).
314 int direction = 1;
315 if ((pjoy->grfbtnPressed & PAD_L2) != _NOT_PRESSED)
316 {
317 // L2 pressed, previous powerup.
318 direction = -1;
319 }
320
321 SetJoyBtnHandled(pjoy, (ushort)(PAD_L2 | PAD_R2));
322
323 uint grfValue;
324 int iNew = iCur;
325 int attemptsLeft = iMax;
326 do
327 {
328 attemptsLeft--;
329 iNew += direction;
330 if (attemptsLeft < 0)
331 {
332 return iCur;
333 }
334 if (iNew < 0)
335 {
336 iNew = iMax - 1;
337 }
338 else if (iMax <= iNew)
339 {
340 iNew = 0;
341 }
342 grfValue = GetGrfvault_unknown();

344
345 // Return the new powerup index.
346 return iNew;
