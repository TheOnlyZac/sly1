6void InitBei(BEI *pbei, CLQ *pclq, float duWidth, float dgHeight, int cseg)
7{
8 // Copy CLQ
9 *(qword *)&pbei->clq = *(qword *)pclq;
10
11 pbei->cseg = cseg;
12
13 // Calculate segment count of half the notch
14 float duSegWidth = 1.0f / cseg;
15 float csegNotchWidth = duWidth / duSegWidth;
16 int csegNotchHalf = (int)csegNotchWidth >> 1; // Simplifies to duWidth * cseg / 2
17
18 pbei->csegNotchHalf = csegNotchHalf;
19
20 // Calculate notch segment indices
21 pbei->isegNotchMid = cseg / 2;
22 pbei->isegNotchFirst = pbei->isegNotchMid - csegNotchHalf;
23 pbei->isegNotchLast = pbei->isegNotchMid + csegNotchHalf;

30float GEvaluateBei(BEI *pbei, int iseg)
31{
32 // A more idiomatic structure would check if inside the notch first
33 // and use the CLQ evaluation as the default return value.
34
35 // Outside notch region
36 if (iseg < pbei->isegNotchFirst || iseg > pbei->isegNotchLast)
37 {
38 // Evaluate quadratic curve
39 float t = iseg / pbei->cseg;
40 return GEvaluateClq(&pbei->clq, t);
41 }
42
43 // Notch edges
44 if (iseg == pbei->isegNotchFirst || iseg == pbei->isegNotchLast)
45 {
46 // Return edge height
47 return pbei->gNotchEdge;
48 }
49
50 // Left side of notch
51 if (iseg < pbei->isegNotchMid)
52 {
53 // Interpolate between edge and center
54 float t = (iseg - pbei->isegNotchFirst) / pbei->csegNotchHalf;
55 return GLerp(pbei->gNotchEdge, pbei->gNotchCenter, t);
56 }
57
58 // Right side of notch
59 if (iseg > pbei->isegNotchMid)
60 {
61 // Interpolate between center and edge
62 float t = (iseg - pbei->isegNotchMid) / pbei->csegNotchHalf;
63 return GLerp(pbei->gNotchCenter, pbei->gNotchEdge, t);
64 }
65
66 return pbei->gNotchCenter;
67}
