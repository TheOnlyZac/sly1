67void FitLinearFunction(float x0, float y0, float x1, float y1, float *pdu, float *pru)
68{
69 if (FFloatsNear(x0, x1, 0.0001f))
70 {
71 *pru = 0.0f;
72 *pdu = y0;

81void FitRecipFunction(float x0, float y0, float x1, float y1, float *pdu, float *pru)
82{
83 float gEpsilon = 0.0001f;
84 if (FFloatsNear(x0, x1, gEpsilon) || __builtin_fabsf(x0) < gEpsilon || __builtin_fabsf(x1) < gEpsilon)
85 {
86 *pru = 0.0f;
87 *pdu = y0;
88 return;

90
91 float deltaX = x0 - x1;
92 float deltaY = y1 - y0;
93
94 float r = ((x0 * x1) * deltaY) / deltaX;

106void ConvertAngleStrength(float deg0, float g0, float deg1, float g1, float *pdu, float *pru)
107{
108 // NOTE: The constant kDegToHalfRad is (3.1415926f / 180.0f) * 0.5f
109 // Putting that here doesn't yield the same output due to precision loss.
110 const float kDegToHalfRad = 0.008726647f;
111 const float kStrengthScale = 1.0f / 255.0f;
112
113 float strength0 = cosf(deg0 * kDegToHalfRad);
114 strength0 += strength0 * strength0 * strength0;
115
116 float strength1 = cosf(deg1 * kDegToHalfRad);
117 strength1 += strength1 * strength1 * strength1;
118
119 FitLinearFunction(strength0, g0 * kStrengthScale, strength1, g1 * kStrengthScale, pdu, pru);
120}
