172int CalculatePercentCompletion(GS *pgs)
173{
174 int cTasksChecked = 0;
175 int cTasksCompleted = 0;
176
177 // Iterate over all the levels in the PchzLevel table
178 for (int i = 0; i < 0x2e; i++)
179 {
180 int levelId = *(int*)&((pchzLevelTable[0].level_id)) + i * sizeof(PchzLevel);
181 int world = levelId >> 8;
182
183 // if world is part of Intro (ie. Splash, Paris, Hideout), skip it
184 if (world != static_cast<int>(GAMEWORLD_Intro))
185 {
186 // get all tasks for the current level
187 int levelTasks = static_cast<int>(pchzLevelTable[0].tasks) + i;
188
189 // get save data for the current level
190 LS* currLs = pgs->aws[world].als + (levelId & 0xff);
191 int currFls = (int)(currLs->fls);
192
193 // check if the level is visited
194 cTasksChecked++;
195 cTasksCompleted = cTasksCompleted + (currFls & (int)(FLS_Visited));
196
197 /* Loop over the bits in the FLS cmp and count how many are set,
198 * but only if those bits are also set in the level_tasks the pchz table */
199 int flsMask = static_cast<int>(FLS_KeyCollected);
200 int tasksToCheck = (int)(levelTasks) & (int)(FLS_KeyCollected);
201 while ((flsMask & ((int)(FLS_KeyCollected) | (int)(FLS_Secondary) | (int)(FLS_Tertiary))) != 0)
202 {
203 if (tasksToCheck != 0)
204 {
205 cTasksChecked++;
206 if ((currFls & flsMask) != 0)
207 {
208 cTasksCompleted++;
209 }
210 }
211 flsMask = flsMask << 1;
212 tasksToCheck = levelTasks & flsMask;
213 }
214 }
215 }
216
217 FWS* pCurrFws = &pgs->aws[1].fws;
218 int i = 4;
219 while (i > -1)
220 {
221 FWS fws_cmp = static_cast<FWS>(*pCurrFws);
222 cTasksChecked++;
223 pCurrFws += 1;
224 i--;
225 if (((int)(fws_cmp) & 0x20) != 0)
226 {
227 cTasksCompleted++;
228 }
229 }
230
231 /* This check ensures we only calculate the % if we have to.
232 *
233 * If cTasksCompleted is 0, finalPercent is left as 0 and
234 * the if block is skipped.
235 *
236 * If cTasksCompleted == cTasksChecked (ie. all tasks are completed),
237 * finalPercent is set to 100 and the if block is skipped.
238 *
239 * In all other cases, the if block sets finalPercent to a calculated value.
240 */
241 int finalPercent = 0;
242 if ((cTasksCompleted != 0) &&
243 (finalPercent = 100, cTasksCompleted != cTasksChecked))
244 {
245 // if cTasksChecked is 0 then something has gone wrong
246 if (cTasksChecked == 0)
247 {
248 return -1;
249 }
250
251 // convert the number of tasks completed to a value between 1 and 100
252 int percent = (cTasksCompleted * 100) / cTasksChecked;
253
254 /* This check accounts for integer division errors.
255 *
256 * If cTasksChecked is <= 0, finalPercent is left as 1 and the
257 * if block is skipped because we already know we have completed at least
258 * one task.
259 *
260 * If the calclated percent is >= 100, finalPercent is set to 99 and the
261 * if block is skipped because we already know all tasks are not complete.
262 *
263 * In all other cases, finalPercent is set to the calculated percent.
264 */
265 finalPercent = 1;
266 if ((cTasksChecked > 0) &&
267 (finalPercent = 99, percent < 100))
268 {
269 finalPercent = percent;
270 }
271 }
272
273 return finalPercent;
274}
